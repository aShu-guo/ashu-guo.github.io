export const UUIDWithSwapTransform: ValueTransformer = {
    to: (uuid: string) => {
        return uuid
            ? Buffer.from(
                uuid.slice(14, 18) + uuid.slice(9, 13) + uuid.slice(0, 8) + uuid.slice(19, 23) + uuid.slice(-12),
                'hex',

                'hex',
                8,
                10,
            )}-${bin.toString('hex', 10, 16)}`
            : bin.toString('hex');
    },
};
