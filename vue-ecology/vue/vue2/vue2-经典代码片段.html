Cache

/*
 зј“еӯҳfnжү§иЎҢзҡ„еҖјпјҢйҖӮз”ЁдәҺеҚ•дёҖе…ҘеҸӮгҖҒзәҜеҮҪж•°
 е°ҶеҶ…еӯҳдёӯзҡ„еҖјдёҺе…ҘеҸӮжҜ”иҫғпјҢеҰӮжһңзӣёеҗҢеҲҷеҸ–еҶ…еӯҳдёӯзҡ„еҖјпјҢеҸҚд№Ӣжү§иЎҢеҮҪж•°еҸ–еҖј
 cacheз»“жһ„пјҡ{е…ҘеҸӮ1:еҮәеҸӮ1, е…ҘеҸӮ2:еҮәеҸӮ2}
*/
function cached (fn) {
        var cache = Object.create(null);
        return (function cachedFn (str) {
            var hit = cache[str];
            return hit || (cache[str] = fn(str))
        })
    }

//ex:
function hello(name){return name}
let cachedFn=cached(hello)
cachedFn('tom tim')
cachedFn('tom tim') // зј“еӯҳеҗҺдјҡзӣҙжҺҘеҸ–еҖјпјҢдёҚеҶҚжү§иЎҢfnпјҢжҸҗй«ҳжҖ§иғҪ
